  <Tab title="NAV Component">
    ## Navigation Bar

    The `NAV` component provides a responsive navigation menu with hamburger toggle and optional explorer button.

    ### Component Code

    ```tsx title="src/components/nav.tsx"
import { NavLink } from "react-router-dom"
import { useState, useEffect, useRef } from "react"
import NavStyles from './../styles/nav.module.css'
import './../styles/nav.module.css'

export default function NavBar({showExplorerButton = true}) {
    const [isNavBarOpen, setIsNavBarOpen] = useState(false)
    const navRef = useRef<HTMLElement>(null)
    const buttonRef = useRef<HTMLButtonElement>(null)

    const navToggle = isNavBarOpen
        ? 'nav open'
        : 'nav'

    const toggleMenu = () => {
        setIsNavBarOpen(!isNavBarOpen)
    }

    useEffect(() => {
        const handleClickOutside = (event: MouseEvent) => {
            if (isNavBarOpen && navRef.current && !navRef.current.contains(event.target as Node) && buttonRef.current && !buttonRef.current.contains(event.target as Node)) {
                setIsNavBarOpen(false);
            }
        }

        document.addEventListener("mousedown", handleClickOutside);
        return () => {
            document.removeEventListener("mousedown", handleClickOutside);
        }
    }, [isNavBarOpen])

    return (
        <div className={NavStyles.NavBar}>
            <div className={NavStyles.menu}>
                <button ref={buttonRef} onClick={toggleMenu} className={NavStyles.button}>
                    <svg xmlns="http://www.w3.org/2000/svg" height="40px" viewBox="0 -960 960 960" width="40px" fill="#000">
                        <path d="M120-240v-80h720v80H120Zm0-200v-80h720v80H120Zm0-200v-80h720v80H120Z"/>
                    </svg>
                </button>
                {showExplorerButton && (
                    <NavLink to='/explorer' className={NavStyles.Explorer_button}>
                        Go to Explorer
                    </NavLink>
                )}
            </div>
            <nav ref={navRef} className={navToggle}>
                <NavLink to='/' onClick={toggleMenu}>Home</NavLink>
                <NavLink to='https://thesimpsonsapi.com/' target="_blank" rel="noopener noreferrer" onClick={toggleMenu}>Api</NavLink>
            </nav>
        </div>
    )
}
    ```

    ### Props

    <ParamField path="showExplorerButton" type="boolean" default="true">
      Controls visibility of the "Go to Explorer" button
    </ParamField>

    ### Features

    - **Responsive hamburger menu** - Toggle navigation on mobile devices
    - **Click-outside detection** - Automatically closes menu when clicking outside
    - **React Router integration** - Uses NavLink for active route styling
    - **External links** - Links to The Simpsons API documentation
    - **Ref management** - Uses refs to track menu and button DOM elements

    ### Usage Example

    ```tsx
import { NAV } from '../components/index.ts'

function MyPage() {
  return (
    <main>
      <NAV />
      {/* Page content */}
    </main>
  )
}
    ```

    <Tip>
    The NAV component handles its own state management, so you don't need to manage the menu open/close state in parent components.
    </Tip>
  </Tab>

  <Tab title="PAGE_COUNTER Component">
    ## Pagination Component

    The `PAGE_COUNTER` component provides pagination controls with flexible layout options (column or row).

    ### Component Code

    ```tsx title="src/components/ui/Page.Counter.tsx"
import CardStyle from './../../styles/pageCounter.module.css';
import type { PageCounterProps } from '../../types.d.ts';

export default function PAGE_COUNTER({ handleNextPage, handlePrevPage, page = 1, totalPages, showtitlePage = true, title, col = true, row = false }: PageCounterProps) {
    return(
        <div className={CardStyle.header}>
            { showtitlePage && (
                <h1 className={CardStyle.title}>{title}</h1>
            )
            }
            { col && (
                <>
                    <div className={CardStyle.buttons}>
                        <button onClick={handlePrevPage} className={CardStyle.button} disabled={page === 1}>
                            Previous page
                        </button>
                        <button onClick={handleNextPage} className={CardStyle.button} disabled={page === totalPages}>
                            Next Page
                        </button>
                    </div>
                    <div className={CardStyle.pagination}>
                        <p>Page {page} of {totalPages}</p>
                    </div>
                </>
            )}
            { row && (
                    <div className={CardStyle.buttons_bottom}>
                        <button onClick={handlePrevPage} className={CardStyle.button} disabled={page === 1}>
                            Previous page
                        </button>
                        <div className={CardStyle.pagination_bottom}>
                            <p>Page {page} of {totalPages}</p>
                        </div>
                        <button onClick={handleNextPage} className={CardStyle.button} disabled={page === totalPages}>
                            Next Page
                        </button>
                    </div>
            )}
        </div>
    )
}
    ```

    ### TypeScript Interface

    ```typescript title="src/types.d.ts"
export interface PageCounterProps {
    handleNextPage?: () => void;
    handlePrevPage?: () => void;
    page: number;
    totalPages: number;
    showtitlePage: boolean;
    title: string | null;
    row: boolean;
    col: boolean;
}
    ```

    ### Props

    <ParamField path="handleNextPage" type="() => void" optional>
      Callback function for next page navigation
    </ParamField>

    <ParamField path="handlePrevPage" type="() => void" optional>
      Callback function for previous page navigation
    </ParamField>

    <ParamField path="page" type="number" required>
      Current page number (1-indexed)
    </ParamField>

    <ParamField path="totalPages" type="number" required>
      Total number of pages available
    </ParamField>

    <ParamField path="showtitlePage" type="boolean" required>
      Whether to display the title above pagination controls
    </ParamField>

    <ParamField path="title" type="string | null" required>
      Title text to display (if showtitlePage is true)
    </ParamField>

    <ParamField path="col" type="boolean" default="true">
      Enable column layout (vertical stacking)
    </ParamField>

    <ParamField path="row" type="boolean" default="false">
      Enable row layout (horizontal arrangement)
    </ParamField>

    ### Usage Example

    From `src/pages/charathers.tsx:51-60`:

    ```tsx
<PAGE_COUNTER
  col={true}
  row={false}
  title="Popular Characters"
  showtitlePage={true}
  page={page}
  totalPages={totalPages}
  handleNextPage={handleNextPage}
  handlePrevPage={handlePrevPage}
/>
    ```

    <Note>
    The component automatically disables navigation buttons at the boundaries (first/last page).
    </Note>

  <Tab title="IMAGE Component">
    ## Image with Loading State

    The `IMAGE` component displays images with automatic loading state management.

    ### Component Code

    ```tsx title="src/components/ui/ImageWithLoading.tsx"
import { useState } from 'react';
import  Loading  from '../icons/Loading.tsx';
import type { ImageProps } from '../../types.d.ts';

export default function ImageWithLoading({ src, alt, className, showtext = true }: ImageProps) {
  const [imageLoaded, setImageLoaded] = useState(false);
  const handleImageLoad = () => {
    setImageLoaded(true);
  };

  return (
    <>
      {
        !imageLoaded && 
        <Loading
        showtext={showtext}
      />
      }

      <img
        src={src}
        alt={alt}
        className={className}
        onLoad={handleImageLoad}
        style={{ display: imageLoaded ? 'block' : 'none' }}
      />
    </>
  );
};
    ```

    ### TypeScript Interface

    ```typescript title="src/types.d.ts"
export interface ImageProps {
    src: string;
    alt: string;
    className?: string;
    showtext: boolean;
}
    ```

    ### Props

    <ParamField path="src" type="string" required>
      Image source URL
    </ParamField>

    <ParamField path="alt" type="string" required>
      Alternative text for accessibility
    </ParamField>

    <ParamField path="className" type="string" optional>
      CSS class name for custom styling
    </ParamField>

    <ParamField path="showtext" type="boolean" required>
      Whether to show "Loading" text in the loading spinner
    </ParamField>

    ### Usage Example

    From `src/pages/charathers.tsx:69`:

    ```tsx
<IMAGE 
  src={`https://cdn.thesimpsonsapi.com/200${character.portrait_path}`} 
  alt={character.name} 
  showtext={false}
/>
    ```

    ### How It Works

    1. Component renders with `imageLoaded` state initially false
    2. Loading spinner displays while image loads
    3. Image loads in background (hidden with `display: none`)
    4. When `onLoad` fires, `imageLoaded` becomes true
    5. Loading spinner hides, image displays

    <Tip>
    Set `showtext={false}` when using multiple images on the same page to avoid repetitive "Loading" text.
    </Tip>

    ## Loading Spinner

    The `LOADING` component displays an animated SVG spinner with optional loading text.

    ### Component Code

    ```tsx title="src/components/ui/Loading.tsx"
import LoadStyle from './../../styles/loading.module.css';

export default function LOADING({showtext = true}) {
    return (
        <div className={LoadStyle.loading}>
            <div className={LoadStyle.loadingItem}>
                <svg xmlns="http://www.w3.org/2000/svg" width={120} height={120} viewBox="0 0 24 24">
                    <g stroke="#ffda05" strokeWidth={1}>
                        <circle cx={12} cy={12} r={9.5} fill="none" strokeLinecap="round" strokeWidth={3}>
                            <animate attributeName="stroke-dasharray" calcMode="spline" dur="1.5s" keySplines="0.42,0,0.58,1;0.42,0,0.58,1;0.42,0,0.58,1" keyTimes="0;0.475;0.95;1" repeatCount="indefinite" values="0 150;42 150;42 150;42 150">
                            </animate>
                            <animate attributeName="stroke-dashoffset" calcMode="spline" dur="1.5s" keySplines="0.42,0,0.58,1;0.42,0,0.58,1;0.42,0,0.58,1" keyTimes="0;0.475;0.95;1" repeatCount="indefinite" values="0;-16;-59;-59"></animate>
                        </circle>
                        <animateTransform attributeName="transform" dur="2s" repeatCount="indefinite" type="rotate" values="0 12 12;360 12 12">
                        </animateTransform>
                    </g>
                </svg>
                {showtext && (
                    <span className={LoadStyle.text}>Loading</span>
                )}
            </div>
        </div>
    )
}
    ```

    ### Props

    <ParamField path="showtext" type="boolean" default="true">
      Whether to display "Loading" text below the spinner
    </ParamField>

    ### Usage Example

    From `src/pages/charathers.tsx:82`:

    ```tsx
{!loading && Array.isArray(characters) && characters.length > 0 ? (
    characters.map((character, i) => (
        {/* Render character cards */}
    ))
) : (
    <LOADING />
)}
    ```

    ### Features

    - **Animated SVG** - Pure SVG animation with no external dependencies
    - **Brand colors** - Uses Simpsons yellow (`#ffda05`) for the spinner
    - **Smooth animation** - Spline-based easing for natural motion
    - **Responsive** - Scales appropriately on different screen sizes

    <Note>
    The loading spinner uses CSS animations defined in `loading.module.css` for additional styling and centering.
    </Note>

## Component Best Practices

<Accordion title="Type Safety">
  All components use TypeScript interfaces defined in `src/types.d.ts` to ensure type safety. Always import the appropriate type when using these components.

  ```tsx
  import type { PageCounterProps, ImageProps } from '../types.d.ts'
  ```
</Accordion>

<Accordion title="CSS Modules">
  Components use CSS Modules for scoped styling, preventing style conflicts:
  
  - `nav.module.css` - Navigation styles
  - `pageCounter.module.css` - Pagination styles
  - `loading.module.css` - Loading spinner styles
</Accordion>

<Accordion title="State Management">
  Components manage their own internal state when appropriate (like NAV's menu toggle) but accept callbacks for external state changes (like PAGE_COUNTER's page handlers).
</Accordion>
